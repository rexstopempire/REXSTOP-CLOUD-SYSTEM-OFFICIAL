Installation Guide

Ensure your Google Script matches the structure below, and you deploy it with "Execute as Me" and "Who has access: Anyone".

const SHEET_ID = "PASTE_YOUR_SHEET_ID_HERE";
function doGet() {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const data = {};
  const sheets = ["Sales", "Warranty", "Debtors", "StockCount", "Overheads", "Payouts"];
  sheets.forEach(name => {
    let sheet = ss.getSheetByName(name);
    if (!sheet) sheet = ss.insertSheet(name);
    const values = sheet.getDataRange().getDisplayValues();
    if (values.length <= 1) { data[name] = []; } else {
      const headers = values[0];
      data[name] = values.slice(1).map(row => {
        const obj = {}; headers.forEach((h, i) => { obj[h] = row[i]; }); return obj;
      });
    }
  });
  return ContentService.createTextOutput(JSON.stringify({ status: "success", data })).setMimeType(ContentService.MimeType.JSON);
}
function doPost(e) {
  try {
    const payload = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.openById(SHEET_ID);
    for (const sheetName in payload) {
      let sheet = ss.getSheetByName(sheetName);
      if (!sheet) sheet = ss.insertSheet(sheetName);
      sheet.clear();
      const records = payload[sheetName];
      if (records.length > 0) {
        const headers = Object.keys(records[0]); sheet.appendRow(headers);
        const rows = records.map(rec => headers.map(h => rec[h] !== undefined ? rec[h] : ""));
        sheet.getRange(2, 1, rows.length, headers.length).setValues(rows);
      }
    }
    return ContentService.createTextOutput(JSON.stringify({ status: "success" })).setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ status: "error", message: err.toString() })).setMimeType(ContentService.MimeType.JSON);
  }
}
